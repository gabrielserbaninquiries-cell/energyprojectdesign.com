# document_engine.py
from docx import Document

def replace_text_in_doc(input_path, output_path, data):
    doc = Document(input_path)

    for p in doc.paragraphs:
        for key, value in data.items():
            if f"<{key}>" in p.text:
                p.text = p.text.replace(f"<{key}>", value)

    doc.save(output_path)

# stamp_engine.py
import os
from docx import Document

STAMP_MARKER = "STAMPILA_GALBENA"

def replace_stamp(doc_path, stamp_image_path):
    doc = Document(doc_path)

    for p in doc.paragraphs:
        if STAMP_MARKER in p.text:
            p.text = ""
            run = p.add_run()
            run.add_picture(stamp_image_path, width=1000000)

    doc.save(doc_path)

# email_service.py
import smtplib
from email.message import EmailMessage

def send_email(to, subject, body, attachment):
    msg = EmailMessage()
    msg['Subject'] = subject
    msg['From'] = "app@energy.ro"
    msg['To'] = to
    msg.set_content(body)

    with open(attachment, 'rb') as f:
        msg.add_attachment(f.read(), maintype='application', subtype='pdf', filename='document.pdf')

    with smtplib.SMTP('smtp.gmail.com', 587) as s:
        s.starttls()
        s.login("EMAIL", "PAROLA")
        s.send_message(msg)
